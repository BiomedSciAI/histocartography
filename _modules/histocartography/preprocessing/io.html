[docs]class DGLGraphLoader(FileLoader):
    def _process(  # type: ignore[override]
        self, path: Union[str, Path]
    ) -> dgl.DGLGraph:
        graph_path = str(path)  # DGL cannot handle pathlib.Path
        graphs, _ = load_graphs(graph_path)
        if len(graphs) == 1:
            return graphs[0]
        return graphs




[docs]class H5Loader(FileLoader):
    def _process(  # type: ignore[override]
        self, path: Union[str, Path]
    ) -> Any:
        with h5py.File(path, "r") as f:
            keys = list(f.keys())
            if len(keys) == 1:
                return h5_to_numpy(f[keys[0]])
            else:
                out = tuple([h5_to_numpy(f[key]) for key in keys])
                return out
